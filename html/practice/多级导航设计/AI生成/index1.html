二级导航菜单实现

基于JavaScript的动态菜单系统

实现说明

  • 使用提供的 data 数组作为菜单数据源
  • 左侧显示一级菜单（主菜单），右侧显示二级菜单（子菜单）
  • 点击有子菜单的项时，右侧显示对应的二级菜单
  • 使用提供的图标资源，通过CDN加载
  • 实现简洁现代的UI设计，响应式布局

JavaScript 实现代码

// 菜单数据 let data = [ { title: '新建', icon: 'https://style.youkeda.com/img/pizza/context-menu/new.png', children: [ { title: '文件', icon: 'https://style.youkeda.com/img/pizza/context-menu/new_file.png' }, { title: '文件夹', icon: 'https://style.youkeda.com/img/pizza/context-menu/new_folder.png' } ] }, { title: '导入', icon: 'https://style.youkeda.com/img/pizza/context-menu/import.png' }, { title: '重命名', icon: 'https://style.youkeda.com/img/pizza/context-menu/rename.png' }, { title: '删除', icon: 'https://style.youkeda.com/img/pizza/context-menu/delete.png' } ]; // 获取DOM元素 const firstMenuBox = document.querySelector('.first'); const secondMenuBox = document.querySelector('.second'); // 清空容器 firstMenuBox.innerHTML = ''; secondMenuBox.innerHTML = ''; // 添加一级菜单标题 const firstTitle = document.createElement('div'); firstTitle.className = 'menu-title'; firstTitle.textContent = '主菜单'; firstMenuBox.appendChild(firstTitle); // 添加二级菜单标题 const secondTitle = document.createElement('div'); secondTitle.className = 'menu-title'; secondTitle.textContent = '子菜单'; secondMenuBox.appendChild(secondTitle); // 渲染一级菜单 for (let i = 0; i < data.length; i++) { const item = data[i]; // 创建菜单项 const menuItem = document.createElement('div'); menuItem.className = 'menu-item'; // 添加图标 const icon = document.createElement('img'); icon.src = item.icon; icon.alt = item.title; // 添加标题 const title = document.createElement('span'); title.textContent = item.title; // 组装菜单项 menuItem.appendChild(icon); menuItem.appendChild(title); // 如果有子菜单，添加箭头图标 if (item.children) { const arrow = document.createElement('i'); arrow.className = 'fas fa-chevron-right'; arrow.style.marginLeft = 'auto'; arrow.style.fontSize = '12px'; menuItem.appendChild(arrow); } // 添加点击事件 menuItem.addEventListener('click', function(e) { e.stopPropagation(); // 移除所有active状态 document.querySelectorAll('.menu-item').forEach(function(el) { el.classList.remove('active'); }); // 设置当前项为active this.classList.add('active'); // 处理子菜单 if (item.children) { renderSubMenu(item.children); } else { secondMenuBox.innerHTML = ''; secondMenuBox.appendChild(secondTitle); // 显示空状态 const emptyState = document.createElement('div'); emptyState.className = 'empty-state'; emptyState.innerHTML = `

此菜单项无子选项

`; secondMenuBox.appendChild(emptyState); } }); // 添加到一级菜单 firstMenuBox.appendChild(menuItem); } // 渲染子菜单函数 function renderSubMenu(children) { // 清空二级菜单（保留标题） secondMenuBox.innerHTML = ''; secondMenuBox.appendChild(secondTitle); // 添加子菜单项 for (let j = 0; j < children.length; j++) { const child = children[j]; const subItem = document.createElement('div'); subItem.className = 'submenu-item'; // 添加图标 const subIcon = document.createElement('img'); subIcon.src = child.icon; subIcon.alt = child.title; // 添加标题 const subTitle = document.createElement('span'); subTitle.textContent = child.title; // 组装子菜单项 subItem.appendChild(subIcon); subItem.appendChild(subTitle); // 添加点击事件 subItem.addEventListener('click', function() { alert(`已选择: ${child.title}`); }); // 添加到二级菜单 secondMenuBox.appendChild(subItem); } }
